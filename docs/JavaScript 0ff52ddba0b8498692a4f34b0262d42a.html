JavaScript

  • JavaScript ist eine Skriptsprache, welche hauptsächlich im Browser Frontend verwendet, kann aber auch im Frontend betrieben werden
  • Macht Seiten dynamisch
  • Basiert auf ECMASCRIPT, welche die Grundlage von JavaScript bzw die Syntax definiert
  • Es gibt verschiedene Versionen, wir verwenden ES6
    • Es gäbe zwar neuere Versionen, werden jedoch nicht überall unterstützt
  • JavaScript unterscheidet sich zu Python unter anderem darin, dass man den ganzen Code auf einer Zeile schreiben könnte. (Es hat Semikolons um Zeilenenden anzuzeigen)
  • Um ein JavaScript im HTML auszuführen, muss ein .js script importiert werden

<script src="js/script.js"></script>

  • JavaScript kann auch unten im body importiert werden, um sicherzustellen, dass alles geladen ist bevor das Skript ausgeführt wird
  • Alternativ kann man node.js installieren und nach npm install JavaScript direkt lokal in der Konsole ausführen

class Car {
    constructor(name, year) {
        this.name = name;
        this.year = year;
        this.kilometer = 0;
    }

    drive(amount) { // methoden müssen nicht definiert werden.

        console.log("Drove " + this.kilometer + "km far.");
    }
}

// einfache Funktion um ein h1 element mit der id "main_heading" farblich zu ändern
function changeHeadingColor() {
    let heading = document.getElementById("main_heading");
    heading.style.color = heading.style.color === "red" ? "black" : "red";
}

Async

// Async -> Asynchroner Code läuft parallel zum Main Code (Main thread), 
// wenn die Resosurcen vorhanden sind

function dosmth(){
    return new Promise(function (fulfill,reject){ // hier wird ein neuer Promise definiert
      // Promises haben immer eine Funktion als Parameter, welche selber mindestens ein Parameter hat 
			// Normalerweise sind das die Namen der "success" und "failure" Funktionen
			setTimeout(function (){fulfill("test");},3000)
			// hier wird nach einem timer von 3s die "success" funktion "fulfill" ausgeführt
			// später (im "then") bedeutet das, dass der Promise "erfüllt" wurde
    })
}
let a=dosmth() // a wird dem Promise gleichgesetzt
a.then(function (value){ // wenn nun die fulfill Funktion aufgerufen wird, wird "then"
    console.log(value) // ausgeführt. Es macht einen log mit "test" 
})
console.log("a: ") // Da ein Promise asynchron ist, läuft dieser Code vor dem "then",
console.log(a) // weil das then einen 3s Timer hat. hier ist der Promise also noch pending
//Promise { <pending> }
